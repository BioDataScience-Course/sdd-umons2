library(MASS) data(wtloss) plot(wtloss) wtlost <- function(t, b0, b1, th) b0 + b1 * 2^(-t/th) wtlost.start <- c(b0 = 100, b1 = 85, th = 100) wtloss.reg <- nls(Weight ~ wtlost(Days, b0, b1,th), data = wtloss, start = wtlost.start, trace = TRUE) summary(wtloss.reg) # Superposer la courbe sur le graphe wtlost2 <- list(Days = seq(from = 0, to = 250, by = 1)) lines(wtlost2$Days, predict(wtloss.reg, newdata = wtlost2), col = “red”)
